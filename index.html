                    /*
                    * Usando un array tomamos las variables que son parte de la funcion useState

                    const[nombre, setNombre] = useState('');

                    // Acceder a la variable o contenido del state

                    // Actualizar o hacer un set al la variable
                    setNombre('Aprendiendo JS'); // actualizar

                    // Ejemplo usando el estate de una clase

                    // Mostrar el contenido del state nombre

                    // Actualizar el state nombre usando setState
                    this.setState({ nombre: 'Aprendiendo JS' });

                    // Mostrar el nuevo contenido
                    console.log(`Nuevo valor: ${this.state.nombre}`);
